function sortLink(head) {
  function mergeSort(p) {
    if (!p || !p.next) {
      return p
    }

    // 寻找中点
    let slow = p
    let fast = p.next
    while (fast && fast.next) {
      fast = fast.next.next
      slow = slow.next
    }

    const mid = slow.next
    slow.next = null
    // 归并排序
    let left = mergeSort(p)
    let right = mergeSort(mid)
    let res = cur = { next: null }
    while (left && right) {
      if (left.value < right.value) {
        cur.next = left
        cur = cur.next
        left = left.next
      } else {
        cur.next = right
        cur = cur.next
        right = right.next
      }
    }
    if (left) {
      cur.next = left
    }
    if (right) {
      cur.next = right
    }
    return res.next
  }

  return mergeSort(head)
}
